import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from linearmodels.panel import PanelOLS

# ── 1. Data preparation ──────────────────────────────────────────────────────
# Entity FE absorb 'treated' (time-invariant) → DO NOT include it as a regressor
# Time FE absorb 'post' → DO NOT include it as a regressor either
# Only the interaction varies within units over time — that's our coefficient
df['treated_post'] = df['treated'] * df['post']

# ── 2. Main DiD estimate ─────────────────────────────────────────────────────
panel = df.set_index(['store', 'month'])

mod = PanelOLS.from_formula(
    'sales ~ treated_post + EntityEffects + TimeEffects',
    data=panel
)
res = mod.fit(cov_type='clustered', cluster_entity=True)
print(res.summary)

att   = res.params['treated_post']
ci_lo = res.conf_int().loc['treated_post', 'lower']
ci_hi = res.conf_int().loc['treated_post', 'upper']
print(f"\n── ATT: {att:.3f}  (95% CI: [{ci_lo:.3f}, {ci_hi:.3f}])")
print(f"   p-value: {res.pvalues['treated_post']:.4f}")

# ── 3. Event study ───────────────────────────────────────────────────────────
TREATMENT_MONTH = 12
REF_PERIOD      = -1      # month 11 (one period before treatment)

def ttp_col(p):
    """Clean column name for time-to-treatment dummy."""
    return f'ttp_m{abs(p)}' if p < 0 else f'ttp_{p}'

df['time_to_treat'] = df['month'] - TREATMENT_MONTH

# Create dummies only for treated stores at each relative period
# Control stores contribute 0s → comparison handled via time FE
periods = sorted([p for p in df['time_to_treat'].unique() if p != REF_PERIOD])
for p in periods:
    df[ttp_col(p)] = ((df['treated'] == 1) & (df['time_to_treat'] == p)).astype(int)

ttp_cols   = [ttp_col(p) for p in periods]
formula_es = 'sales ~ ' + ' + '.join(ttp_cols) + ' + EntityEffects + TimeEffects'

res_es = PanelOLS.from_formula(formula_es, data=df.set_index(['store', 'month'])) \
                 .fit(cov_type='clustered', cluster_entity=True)

# Reconstruct full coefficient series (insert 0 at reference period)
all_periods = sorted(df['time_to_treat'].unique())
coefs, lowers, uppers = [], [], []
for p in all_periods:
    if p == REF_PERIOD:
        coefs.append(0); lowers.append(0); uppers.append(0)
    else:
        col = ttp_col(p)
        coefs.append(res_es.params[col])
        lowers.append(res_es.conf_int().loc[col, 'lower'])
        uppers.append(res_es.conf_int().loc[col, 'upper'])

# ── Plot ─────────────────────────────────────────────────────────────────────
fig, ax = plt.subplots(figsize=(12, 5))
coefs_arr  = np.array(coefs)
lowers_arr = np.array(lowers)
uppers_arr = np.array(uppers)

ax.fill_between(all_periods, lowers_arr, uppers_arr,
                alpha=0.15, color='steelblue')
ax.plot(all_periods, coefs_arr,
        'o-', color='steelblue', linewidth=1.8, markersize=5, label='Event study coef')
ax.axhline(0,    color='black',  linewidth=0.8, linestyle='--')
ax.axvline(-0.5, color='red',    linewidth=1.2, linestyle=':',
           label='Treatment onset (month 12)')

# Shade pre-period
ax.axvspan(min(all_periods) - 0.5, -0.5, alpha=0.04, color='gray')

ax.set_xlabel('Months Relative to Treatment')
ax.set_ylabel('Effect on Sales')
ax.set_title('Event Study: New Inventory System → Sales')
ax.legend(); plt.tight_layout(); plt.show()

# ── 4. Joint pre-trends test ─────────────────────────────────────────────────
pre_cols = [ttp_col(p) for p in periods if p < 0]
print("\n── Pre-treatment coefficients (should be near zero):")
print(pd.DataFrame({
    'coef':    res_es.params[pre_cols].round(4),
    'p-value': res_es.pvalues[pre_cols].round(4)
}))

print(df.groupby('treatment')['ps'].describe().round(4))

fig, ax = plt.subplots(figsize=(8, 4))

#   Treated:  w = 1 / ps
#   Control:  w = 1 / (1 - ps)
df['ipw_weight'] = np.where(
    df['treatment'] == 1,
    1.0 / df['ps'],
    1.0 / (1.0 - df['ps'])
)

# Stabilized weights (reduces variance from extreme scores)
p_treat = df['treatment'].mean()
df['ipw_weight_stab'] = np.where(
    df['treatment'] == 1,
    p_treat / df['ps'],
    (1 - p_treat) / (1.0 - df['ps'])
)

print(f"\nRaw IPW weight summary:")
print(df['ipw_weight'].describe().round(4))
print(f"\nStabilized IPW weight summary:")

def smd(treated_vals, control_vals, weights_control=None, weights_treated=None):
    """Standardized mean difference (pooled SD from unweighted sample)."""
    pooled_sd = np.std(
        np.concatenate([treated_vals, control_vals]), ddof=1
    )
    m_t = (np.average(treated_vals, weights=weights_treated)
           if weights_treated is not None else treated_vals.mean())
    m_c = (np.average(control_vals, weights=weights_control)
           if weights_control is not None else control_vals.mean())
    return (m_t - m_c) / pooled_sd if pooled_sd > 0 else 0.0

balance_rows = []

    c_vals = df.loc[df['treatment'] == 0, cov].values

    t_w    = df.loc[df['treatment'] == 1, 'ipw_weight_stab'].values

    balance_rows.append({

        'SMD_before': round(smd(t_vals, c_vals), 4),
        'SMD_after':  round(smd(t_vals, c_vals, c_w, t_w), 4),

balance_df = pd.DataFrame(balance_rows)
balance_df['balanced'] = balance_df['SMD_after'].abs() < 0.1

print(balance_df.to_string(index=False))

if not balance_df['balanced'].all():

print(f"  p-value:               {model_ipw.pvalues['treatment']:.4f}")
print(f"{'─'*45}")

print(f"ESTIMATE:{ate:.4f}")
