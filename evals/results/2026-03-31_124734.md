n_treat = len(treat_conv)
n_ctrl  = len(ctrl_conv)

# Pooled standard error for the difference in proportions
se_diff = np.sqrt(p_treat * (1 - p_treat) / n_treat +
                  p_ctrl  * (1 - p_ctrl)  / n_ctrl)

ate     = p_treat - p_ctrl
z_stat  = ate / se_diff
p_val   = 2 * (1 - stats.norm.cdf(abs(z_stat)))
ci_low  = ate - 1.96 * se_diff
ci_high = ate + 1.96 * se_diff

# Cohen's h (effect size for proportions)
cohens_h = 2 * np.arcsin(np.sqrt(p_treat)) - 2 * np.arcsin(np.sqrt(p_ctrl))

print(f"  Control   conversion rate: {p_ctrl:.4f}  ({p_ctrl:.2%})")
print(f"  Treatment conversion rate: {p_treat:.4f}  ({p_treat:.2%})")

print(f"  95% CI:                    [{ci_low:+.4f}, {ci_high:+.4f}]")
print(f"  Z-statistic:               {z_stat:.4f}")
print(f"  P-value:                   {p_val:.4f}")
print(f"  Cohen's h (effect size):   {abs(cohens_h):.4f}  "
      f"({'small' if abs(cohens_h) < 0.2 else 'medium' if abs(cohens_h) < 0.5 else 'large'})")

# Covariate-adjusted (recommended: more precise, controls for any imbalance)

)

)
