import numpy as np
from linearmodels.panel import PanelOLS

# ── 1. Construct DiD variables ─────────────────────────────────────────────
df['treated']      = (df['store'] <= 25).astype(int)   # time-invariant group flag
df['post']         = (df['month'] >= 6).astype(int)    # post-treatment indicator
df['treated_post'] = df['treated'] * df['post']        # DiD interaction (varies over time)

# ── 2. Set panel index ─────────────────────────────────────────────────────
df_panel = df.set_index(['store', 'month'])

# ── 3. Estimate TWFE DiD ───────────────────────────────────────────────────
# EntityEffects absorb all time-invariant store characteristics (including
# the 'treated' dummy itself — do NOT add it as a regressor to avoid
# perfect multicollinearity). 'treated_post' is fine because it varies
# over time within treated stores.
mod = PanelOLS.from_formula(
    'sales ~ treated_post + EntityEffects + TimeEffects',
    data=df_panel
)
res = mod.fit(cov_type='clustered', cluster_entity=True)

# ── 4. Extract and report results ──────────────────────────────────────────
att   = res.params['treated_post']
ci    = res.conf_int()
lower = ci.loc['treated_post', 'lower']
upper = ci.loc['treated_post', 'upper']
pval  = res.pvalues['treated_post']

print(res.summary)
print("─" * 50)
print(f"DiD Treatment Effect (ATT) : {att:+.4f}")
print(f"95% Confidence Interval    : [{lower:.4f}, {upper:.4f}]")
print(f"p-value                    : {pval:.4f}")
print(f'ESTIMATE:{att}')
